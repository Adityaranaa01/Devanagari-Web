          {/* Enhanced Selected Ingredients Summary */}
          {selectedIngredients.size > 0 && (
            <div className="mt-20 max-w-6xl mx-auto">
              <div className="bg-white/80 backdrop-blur-sm rounded-3xl p-8 shadow-xl border border-gray-200">
                <h3 className="text-2xl font-bold text-[#4A5C3D] mb-8 text-center">
                  Your Selected Ingredients ({selectedIngredients.size})
                </h3>
                <div className="grid grid-cols-2 md:grid-cols-4 lg:grid-cols-7 gap-4">
                  {Array.from(selectedIngredients).map((ingredientName) => (
                    <div
                      key={ingredientName}
                      className="bg-gradient-to-r from-[#4A5C3D]/10 to-[#6B8E23]/10 px-4 py-3 rounded-xl text-center text-sm font-medium text-[#4A5C3D] border border-[#4A5C3D]/20 hover:from-[#4A5C3D]/20 hover:to-[#6B8E23]/20 transition-all duration-300"
                    >
                      <CheckCircle size={14} className="inline mr-2" />
                      {ingredientName}
                    </div>
                  ))}
                </div>
              </div>
            </div>
          )}
        </div>
      </section>
      
      {/* Enhanced Testimonials Section */}
      <section className="py-32 bg-gradient-to-br from-[#F8F6F3] to-[#EAE2D5] relative overflow-hidden">
        {/* Background Pattern */}
        <div className="absolute inset-0 opacity-20">
          <div className="absolute top-0 left-1/4 w-32 h-32 border-2 border-[#4A5C3D] rounded-full"></div>
          <div className="absolute bottom-1/4 right-1/4 w-48 h-48 border-2 border-[#6B8E23] rounded-full"></div>
          <div className="absolute top-3/4 left-3/4 w-24 h-24 border-2 border-[#CDE0B7] rounded-full"></div>
        </div>
        
        <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 relative z-10">
          {/* Section Header */}
          <div className="text-center mb-20">
            <div className="inline-flex items-center px-4 py-2 rounded-full bg-[#4A5C3D]/10 backdrop-blur-sm border border-[#4A5C3D]/20 mb-6">
              <span className="text-[#4A5C3D] text-sm font-medium tracking-wide">
                CUSTOMER STORIES
              </span>
            </div>
            <h2 className="text-4xl md:text-5xl lg:text-6xl font-bold text-[#4A5C3D] mb-6 leading-tight">
              What Our
              <br />
              <span className="bg-gradient-to-r from-[#4A5C3D] to-[#6B8E23] bg-clip-text text-transparent">
                Customers Say
              </span>
            </h2>
            <div className="h-1 w-24 bg-gradient-to-r from-[#CDE0B7] to-[#F6CFA7] rounded-full mx-auto mb-6"></div>
            <p className="text-xl text-gray-600 max-w-3xl mx-auto leading-relaxed">
              Real stories from real people who've transformed their wellness journey with Devanagari Health Mix.
            </p>
          </div>
          
          {/* Enhanced Testimonials Grid */}
          <div className="grid grid-cols-1 md:grid-cols-3 gap-8">
            {testimonials.map((testimonial, index) => (
              <div key={index} className="group">
                <div className="bg-white rounded-3xl p-8 shadow-xl hover:shadow-2xl transition-all duration-500 group-hover:-translate-y-2 border border-gray-100 relative overflow-hidden">
                  {/* Quote Mark */}
                  <div className="absolute top-6 right-6 text-6xl text-[#CDE0B7]/30 font-serif">"</div>
                  
                  {/* Stars */}
                  <div className="flex items-center mb-6">
                    {[...Array(testimonial.rating)].map((_, i) => (
                      <Star key={i} className="text-[#F6CFA7] fill-current mr-1" size={20} />
                    ))}
                  </div>
                  
                  {/* Quote */}
                  <blockquote className="text-gray-700 mb-8 leading-relaxed text-lg relative z-10">
                    "{testimonial.quote}"
                  </blockquote>
                  
                  {/* Author */}
                  <div className="flex items-center space-x-4">
                    <div className="w-12 h-12 bg-gradient-to-r from-[#4A5C3D] to-[#6B8E23] rounded-full flex items-center justify-center">
                      <span className="text-white font-bold text-lg">
                        {testimonial.name.charAt(0)}
                      </span>
                    </div>
                    <div>
                      <p className="font-bold text-[#4A5C3D] text-lg">
                        {testimonial.name}
                      </p>
                      <p className="text-gray-500 text-sm">
                        {testimonial.location}
                      </p>
                    </div>
                  </div>
                  
                  {/* Decorative elements */}
                  <div className="absolute bottom-6 right-6 w-3 h-3 bg-[#F6CFA7] rounded-full opacity-60"></div>
                  <div className="absolute bottom-8 right-10 w-2 h-2 bg-[#CDE0B7] rounded-full opacity-40"></div>
                </div>
              </div>
            ))}
          </div>

          {/* CTA Section */}
          <div className="mt-20 text-center">
            <div className="bg-white rounded-3xl p-12 shadow-2xl border border-gray-100 max-w-4xl mx-auto">
              <h3 className="text-3xl font-bold text-[#4A5C3D] mb-4">
                Join Thousands of Satisfied Customers
              </h3>
              <p className="text-xl text-gray-600 mb-8 leading-relaxed">
                Start your wellness journey today with Devanagari Health Mix
              </p>
              <div className="flex flex-col sm:flex-row gap-4 justify-center">
                <button className="group bg-gradient-to-r from-[#4A5C3D] to-[#5d7249] text-white px-10 py-5 rounded-2xl text-xl font-bold hover:from-[#3a4a2f] hover:to-[#4a5d3a] transition-all duration-300 flex items-center justify-center space-x-3 shadow-2xl hover:shadow-3xl transform hover:scale-105">
                  <ShoppingBag size={24} className="group-hover:animate-bounce" />
                  <span className="font-bold tracking-wide">Order Now</span>
                </button>
                <button className="group bg-transparent text-[#4A5C3D] px-10 py-5 rounded-2xl text-xl font-medium border-2 border-[#4A5C3D] hover:bg-[#4A5C3D] hover:text-white transition-all duration-300 flex items-center justify-center space-x-3">
                  <span>Learn More</span>
                  <ArrowRight size={24} className="group-hover:translate-x-1 transition-transform" />
                </button>
              </div>
            </div>
          </div>
        </div>
      </section>

      {/* Ingredient Modal */}
      {selectedIngredient && (
        <IngredientModal
          ingredient={selectedIngredient}
          isOpen={isModalOpen}
          onClose={closeModal}
        />
      )}
    </div>
  );
};

export default Home;
